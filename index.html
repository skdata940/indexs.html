// supply-chain-cc — Hyperledger Fabric v2.5
package main

import (
  "encoding/json"
  "fmt"
  "time"
  "github.com/hyperledger/fabric-contract-api-go/contractapi"
)

type SmartContract struct {
  contractapi.Contract
}

type Product struct {
  ID           string                 `json:"id"`
  Name         string                 `json:"name"`
  Category     string                 `json:"category"`
  Owner        string                 `json:"owner"`
  Location     string                 `json:"location"`
  Status       string                 `json:"status"`
  Timestamp    string                 `json:"timestamp"`
  Hash         string                 `json:"hash"`
  Attributes   map[string]string      `json:"attributes"`
}

func (s *SmartContract) RegisterProduct(
  ctx contractapi.TransactionContextInterface,
  productId string, productJSON string,
) error {
  exists, err := s.productExists(ctx, productId)
  if err != nil { return err }
  if exists {
    return fmt.Errorf("product %s already exists", productId)
  }
  mspId, _ := ctx.GetClientIdentity().GetMSPID()
  txTimestamp, _ := ctx.GetStub().GetTxTimestamp()
  
  var product Product
  json.Unmarshal([]byte(productJSON), &product)
  product.Timestamp = time.Unix(txTimestamp.Seconds, 0).Format(time.RFC3339)
  product.Owner = mspId
  
  data, _ := json.Marshal(product)
  ctx.GetStub().SetEvent("ProductRegistered", data)
  return ctx.GetStub().PutState(productId, data)
}

func (s *SmartContract) TransferOwnership(
  ctx contractapi.TransactionContextInterface,
  productId, newOwner, newLocation,
  reason, submitter, timestamp string,
) error {
  product, err := s.QueryProduct(ctx, productId)
  if err != nil { return err }
  product.Owner = newOwner
  product.Location = newLocation
  product.Status = "IN_TRANSIT"
  product.Timestamp = timestamp
  data, _ := json.Marshal(product)
  ctx.GetStub().SetEvent("OwnershipTransferred", data)
  return ctx.GetStub().PutState(productId, data)
}
// Fabric Node.js SDK — ChainTrace API Client
const axios = require('axios');

const API = 'http://localhost:3001/api/v1';
let token; // JWT from /auth/login

// Authenticate
async function login(userId, password) {
  const { data } = await axios.post(`${API}/auth/login`, 
    { userId, password });
  token = data.data.token;
  return data;
}

// Register product on blockchain
async function registerProduct(product, channel = 'supply-chain-01') {
  return axios.post(`${API}/products`, product, {
    headers: {
      Authorization: `Bearer ${token}`,
      'x-channel': channel,
    }
  });
}

// Transfer ownership
async function transferOwnership(productId, newOwner, location) {
  return axios.put(`${API}/products/${productId}/transfer`,
    { newOwner, newLocation: location },
    { headers: { Authorization: `Bearer ${token}` } }
  );
}
